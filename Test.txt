import random
import re

def generate_phone_number_variations(n):
    """
    Generates a list of U.S. phone numbers with exactly 10 digits in various formats.
    
    Args:
        n (int): The number of phone numbers to generate.
    
    Returns:
        list: A list of phone numbers as strings that look like phone numbers and contain exactly 10 digits.
    """
    # Define a regex pattern for generating phone numbers with exactly 10 digits
    patterns = [
        r"\d{3}-\d{3}-\d{4}",       # Format: 123-456-7890
        r"\(\d{3}\) \d{3}-\d{4}",   # Format: (123) 456-7890
        r"\d{3}\.\d{3}\.\d{4}",     # Format: 123.456.7890
        r"\d{10}",                  # Format: 1234567890
        r"\d{3} \d{3} \d{4}",       # Format: 123 456 7890
        r"\d{3}/\d{3}/\d{4}",       # Format: 123/456/7890
        r"\d{3}-\d{3}\.\d{4}",      # Format: 123-456.7890
        r"\d{3}\.\d{3}-\d{4}",      # Format: 123.456-7890
        r"\(\d{3}\)\d{3}-\d{4}",    # Format: (123)456-7890
        r"\d{3}\*\d{3}\*\d{4}",     # Format: 123*234*5678
    ]

    phone_numbers = []

    for _ in range(n):
        # Randomly select a pattern
        pattern = random.choice(patterns)

        # Generate a phone number based on the selected pattern
        phone_number = re.sub(r"\d", lambda x: str(random.randint(0, 9)), pattern)
        phone_numbers.append(phone_number)

    return phone_numbers
